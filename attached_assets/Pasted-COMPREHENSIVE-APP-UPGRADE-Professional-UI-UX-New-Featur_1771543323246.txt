```javascript
async function handleCallHangup(callLog, payload) {
  // ... existing hangup logic ...
  
  // Get campaign settings
  const campaign = await db.query(
    'SELECT * FROM campaigns WHERE id = ?',
    [callLog.campaign_id]
  );
  
  if (campaign && campaign.status === 'running') {
    const pendingContacts = await db.query(
      `SELECT COUNT(*) as count FROM contacts 
       WHERE campaign_id = ? AND status = 'pending'`,
      [callLog.campaign_id]
    );
    
    if (pendingContacts.count > 0) {
      let delaySeconds;
      
      // CRITICAL: Check interval type
      if (campaign.call_interval_type === 'interval') {
        // Use fixed interval mode
        delaySeconds = campaign.call_interval_minutes * 60;
        console.log(`üìû Interval Mode: Waiting ${campaign.call_interval_minutes} minutes before next call`);
      } else {
        // Use pace mode (continuous)
        delaySeconds = Math.ceil(60 / campaign.dialing_pace);
        console.log(`‚ö° Pace Mode: Waiting ${delaySeconds} seconds before next call`);
      }
      
      console.log(`Next call scheduled in ${delaySeconds} seconds (${Math.round(delaySeconds/60)} minutes)`);
      
      // Schedule next call with calculated delay
      setTimeout(() => {
        dialNextNumber(callLog.campaign_id);
      }, delaySeconds * 1000);
      
    } else {
      // Complete campaign
      await db.query(
        `UPDATE campaigns 
         SET status = 'completed', completed_at = ? 
         WHERE id = ?`,
        [new Date(), callLog.campaign_id]
      );
    }
  }
}
```

=== PART 2: FLOATING NOTEPAD BUBBLE ===

REQUIREMENT:
Convert the notepad feature into a floating bubble that can be dragged anywhere on screen and has an option to stick to default position (bottom-right corner).

IMPLEMENTATION:

1. **Create FloatingNotepad Component:**
```jsx
import React, { useState, useEffect, useRef } from 'react';

function FloatingNotepad() {
  const [isOpen, setIsOpen] = useState(false);
  const [isDragging, setIsDragging] = useState(false);
  const [isSticky, setIsSticky] = useState(true); // Default: stuck to corner
  const [position, setPosition] = useState({ x: null, y: null });
  const [notes, setNotes] = useState('');
  const notepadRef = useRef(null);
  const dragOffset = useRef({ x: 0, y: 0 });

  // Load notes from localStorage
  useEffect(() => {
    const savedNotes = localStorage.getItem('notepad_content');
    const savedPosition = localStorage.getItem('notepad_position');
    const savedSticky = localStorage.getItem('notepad_sticky');
    
    if (savedNotes) setNotes(savedNotes);
    if (savedPosition) setPosition(JSON.parse(savedPosition));
    if (savedSticky !== null) setIsSticky(savedSticky === 'true');
  }, []);

  // Save notes to localStorage
  useEffect(() => {
    localStorage.setItem('notepad_content', notes);
  }, [notes]);

  // Save position
  useEffect(() => {
    if (!isSticky && position.x !== null) {
      localStorage.setItem('notepad_position', JSON.stringify(position));
    }
  }, [position, isSticky]);

  // Save sticky state
  useEffect(() => {
    localStorage.setItem('notepad_sticky', isSticky.toString());
  }, [isSticky]);

  // Get default position (bottom-right)
  const getDefaultPosition = () => {
    return {
      x: window.innerWidth - 420, // 400px width + 20px margin
      y: window.innerHeight - 620  // 600px height + 20px margin
    };
  };

  // Calculate current position
  const getCurrentPosition = () => {
    if (isSticky || position.x === null) {
      return getDefaultPosition();
    }
    return position;
  };

  // Handle drag start
  const handleMouseDown = (e) => {
    if (e.target.closest('.notepad-header-drag-handle')) {
      setIsDragging(true);
      const currentPos = getCurrentPosition();
      dragOffset.current = {
        x: e.clientX - currentPos.x,
        y: e.clientY - currentPos.y
      };
    }
  };

  // Handle drag move
  useEffect(() => {
    const handleMouseMove = (e) => {
      if (isDragging) {
        const newX = e.clientX - dragOffset.current.x;
        const newY = e.clientY - dragOffset.current.y;
        
        // Keep within viewport bounds
        const maxX = window.innerWidth - 400;
        const maxY = window.innerHeight - 600;
        
        setPosition({
          x: Math.max(0, Math.min(newX, maxX)),
          y: Math.max(0, Math.min(newY, maxY))
        });
        
        // Automatically unstick when dragging
        if (isSticky) {
          setIsSticky(false);
        }
      }
    };

    const handleMouseUp = () => {
      setIsDragging(false);
    };

    if (isDragging) {
      window.addEventListener('mousemove', handleMouseMove);
      window.addEventListener('mouseup', handleMouseUp);
    }

    return () => {
      window.removeEventListener('mousemove', handleMouseMove);
      window.removeEventListener('mouseup', handleMouseUp);
    };
  }, [isDragging, isSticky]);

  // Toggle sticky mode
  const toggleSticky = () => {
    setIsSticky(!isSticky);
    if (!isSticky) {
      // Going back to sticky - reset position
      setPosition({ x: null, y: null });
    }
  };

  const currentPos = getCurrentPosition();

  return (
    <>
      {/* Floating Bubble Button */}
      {!isOpen && (
        <div 
          className={`notepad-bubble ${isSticky ? 'sticky' : 'floating'}`}
          style={!isSticky && position.x !== null ? {
            left: `${position.x + 350}px`,
            top: `${position.y}px`
          } : {}}
          onClick={() => setIsOpen(true)}
          title="Open Notepad"
        >
          <svg width="24" height="24" viewBox="0 0 24 24" fill="none">
            <path d="M14 2H6a2 2 0 0 0-2 2v16a2 2 0 0 0 2 2h12a2 2 0 0 0 2-2V8l-6-6z" 
                  stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"/>
            <path d="M14 2v6h6M16 13H8M16 17H8M10 9H8" 
                  stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"/>
          </svg>
          <div className="bubble-badge">Notes</div>
        </div>
      )}

      {/* Notepad Window */}
      {isOpen && (
        <div 
          ref={notepadRef}
          className={`floating-notepad ${isDragging ? 'dragging' : ''} ${isSticky ? 'sticky' : 'floating'}`}
          style={!isSticky && currentPos.x !== null ? {
            left: `${currentPos.x}px`,
            top: `${currentPos.y}px`,
            position: 'fixed'
          } : {}}
        >
          {/* Header */}
          <div 
            className="notepad-header"
            onMouseDown={handleMouseDown}
          >
            <div className="notepad-header-drag-handle" title="Drag to move">
              <svg width="20" height="20" viewBox="0 0 24 24" fill="none">
                <circle cx="5" cy="7" r="1.5" fill="currentColor"/>
                <circle cx="12" cy="7" r="1.5" fill="currentColor"/>
                <circle cx="19" cy="7" r="1.5" fill="currentColor"/>
                <circle cx="5" cy="12" r="1.5" fill="currentColor"/>
                <circle cx="12" cy="12" r="1.5" fill="currentColor"/>
                <circle cx="19" cy="12" r="1.5" fill="currentColor"/>
                <circle cx="5" cy="17" r="1.5" fill="currentColor"/>
                <circle cx="12" cy="17" r="1.5" fill="currentColor"/>
                <circle cx="19" cy="17" r="1.5" fill="currentColor"/>
              </svg>
            </div>
            
            <div className="notepad-title">
              <svg width="18" height="18" viewBox="0 0 24 24" fill="none">
                <path d="M14 2H6a2 2 0 0 0-2 2v16a2 2 0 0 0 2 2h12a2 2 0 0 0 2-2V8l-6-6z" 
                      stroke="currentColor" strokeWidth="2"/>
              </svg>
              Quick Notes
            </div>

            <div className="notepad-controls">
              <button 
                className={`sticky-toggle ${isSticky ? 'active' : ''}`}
                onClick={toggleSticky}
                title={isSticky ? 'Unpin from corner' : 'Pin to bottom-right corner'}
              >
                {isSticky ? 'üìå' : 'üìç'}
              </button>
              
              <button 
                className="close-button"
                onClick={() => setIsOpen(false)}
                title="Close (notes are auto-saved)"
              >
                ‚úï
              </button>
            </div>
          </div>

          {/* Content */}
          <div className="notepad-content">
            <textarea
              value={notes}
              onChange={(e) => setNotes(e.target.value)}
              placeholder="Type your notes here... They auto-save automatically."
              className="notepad-textarea"
            />
          </div>

          {/* Footer */}
          <div className="notepad-footer">
            <span className="auto-save-indicator">
              ‚úì Auto-saved
            </span>
            <span className="character-count">
              {notes.length} characters
            </span>
          </div>
        </div>
      )}
    </>
  );
}

export default FloatingNotepad;
```
